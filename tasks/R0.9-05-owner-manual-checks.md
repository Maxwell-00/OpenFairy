# 4. Terminal B：定义验证函数

```powershell
$BaseHttp =
    "http://127.0.0.1:$GatewayPort"

function Get-FairyWorkerProcesses {
    return @(
        Get-CimInstance Win32_Process |
            Where-Object {
                $_.CommandLine -match
                    'mimo_asr_worker\.py|' +
                    'minimax_tts_worker\.py'
            }
    )
}

function Get-FairyGatewayProcesses {
    return @(
        Get-CimInstance Win32_Process |
            Where-Object {
                $_.CommandLine -match
                    'start-gateway\.mjs|' +
                    'apps[\\/]gateway[\\/]' +
                    'src[\\/]bin[\\/]gateway\.ts'
            }
    )
}

function Get-SpeechTempRoots {
    return @(
        Get-ChildItem $env:TEMP `
            -Directory `
            -ErrorAction SilentlyContinue |
            Where-Object {
                $_.Name -match
                    'fairy-.*(?:mimo-asr|minimax-tts)'
            } |
            Select-Object -ExpandProperty FullName
    )
}

function Test-LoopbackPortAvailable {
    param(
        [Parameter(Mandatory)]
        [int]$Port
    )

    $Listener =
        [Net.Sockets.TcpListener]::new(
            [Net.IPAddress]::Loopback,
            $Port
        )

    try {
        $Listener.Start()
        return $true
    } catch {
        return $false
    } finally {
        try {
            $Listener.Stop()
        } catch {
        }
    }
}

function Get-FairySessions {
    $Headers = @{
        Authorization =
            "Bearer $env:FAIRY_GATEWAY_TOKEN"
    }

    $Body =
        Invoke-RestMethod `
            -Uri "$BaseHttp/sessions" `
            -Headers $Headers `
            -Method Get

    return @($Body.sessions)
}

function Wait-NewFairySession {
    param(
        [Parameter(Mandatory)]
        [string[]]$BeforeIds
    )

    for (
        $Attempt = 0;
        $Attempt -lt 100;
        $Attempt += 1
    ) {
        $Current = @(Get-FairySessions)

        $New = @(
            $Current |
                Where-Object {
                    $_.id -notin $BeforeIds
                } |
                Sort-Object `
                    last_active `
                    -Descending
        )

        if ($New.Count -eq 1) {
            return [string]$New[0].id
        }

        if ($New.Count -gt 1) {
            throw 'More than one new session appeared'
        }

        Start-Sleep -Milliseconds 200
    }

    throw 'Timed out waiting for one new session'
}

function Read-SessionEvents {
    param(
        [Parameter(Mandatory)]
        [string]$Sid
    )

    $LogPath =
        Join-Path `
            $DataDir `
            "sessions\$Sid\log.jsonl"

    if (
        -not (
            Test-Path `
                -LiteralPath $LogPath `
                -PathType Leaf
        )
    ) {
        throw "Session log is missing for $Sid"
    }

    return @(
        Get-Content -LiteralPath $LogPath |
            Where-Object {
                $_.Trim().Length -gt 0
            } |
            ForEach-Object {
                $_ | ConvertFrom-Json
            }
    )
}

function Read-ArtifactRecords {
    $RegistryPath =
        Join-Path `
            $DataDir `
            'artifacts\artifacts.jsonl'

    if (
        -not (
            Test-Path `
                -LiteralPath $RegistryPath `
                -PathType Leaf
        )
    ) {
        return @()
    }

    return @(
        Get-Content -LiteralPath $RegistryPath |
            Where-Object {
                $_.Trim().Length -gt 0
            } |
            ForEach-Object {
                $_ | ConvertFrom-Json
            }
    )
}
```

$DoctorWorkersAfter =

# 8. 一次真实正常语音

**成功一次后不要重复。**

浏览器打开后、创建 session 前，在 Terminal B：

```powershell
$BeforeSessionIds = @(
    Get-FairySessions |
        Select-Object -ExpandProperty id
)
```

浏览器中：

1. 输入 gateway token；
2. 创建一个新 session；
3. 录制一次：

```text
请用一句话 summarize today's demo.
```

4. 等待 transcript；
5. 等待 assistant final；
6. 等待 MP3 control；
7. autoplay 被阻止时按 Play；
8. 按一次 Stop；
9. reload session，确认历史重建。

定位新 session：

```powershell
$VoiceSid =
    Wait-NewFairySession `
        -BeforeIds $BeforeSessionIds

$VoiceEvents =
    @(Read-SessionEvents -Sid $VoiceSid)
```

提取计数：

```powershell
$AsrFinals = @(
    $VoiceEvents |
        Where-Object type -eq 'speech.asr.final'
)

$TurnInputs = @(
    $VoiceEvents |
        Where-Object type -eq 'turn.input'
)

$TurnFinals = @(
    $VoiceEvents |
        Where-Object type -eq 'turn.final'
)

$TtsChunks = @(
    $VoiceEvents |
        Where-Object type -eq 'speech.tts.chunk'
)

$ContextManifests = @(
    $VoiceEvents |
        Where-Object type -eq 'context.manifest'
)
```

验证：

```powershell
if ($AsrFinals.Count -ne 1) {
    throw 'Expected one speech.asr.final'
}

if ($TurnInputs.Count -ne 1) {
    throw 'Expected one turn.input'
}

if ($TurnFinals.Count -ne 1) {
    throw 'Expected one turn.final'
}

if ($TtsChunks.Count -ne 1) {
    throw 'Expected one speech.tts.chunk'
}

if ($ContextManifests.Count -ne 1) {
    throw 'Expected one model context manifest'
}

$Transcript =
    [string]$AsrFinals[0].payload.text

$AssistantText = @(
    $TurnFinals[0].payload.content |
        Where-Object kind -eq 'text' |
        ForEach-Object {
            [string]$_.text
        }
) -join "`n"

if ([string]::IsNullOrWhiteSpace($Transcript)) {
    throw 'Transcript is empty'
}

if (
    [string]::IsNullOrWhiteSpace($AssistantText)
) {
    throw 'Assistant final is empty'
}
```

验证 artifacts：

```powershell
$ArtifactRecords =
    @(Read-ArtifactRecords)

$InputRef =
    [string]$AsrFinals[0].payload.audio_ref

$SpeechRef =
    [string]$TtsChunks[0].payload.audio_ref

$InputRecord =
    $ArtifactRecords |
    Where-Object artifact_id -eq $InputRef |
    Select-Object -Last 1

$SpeechRecord =
    $ArtifactRecords |
    Where-Object artifact_id -eq $SpeechRef |
    Select-Object -Last 1

if (-not $InputRecord) {
    throw 'Input artifact is missing'
}

if (-not $SpeechRecord) {
    throw 'Speech artifact is missing'
}

if (
    $InputRecord.kind -ne 'input' -or
    $InputRecord.mime -ne 'audio/wav'
) {
    throw 'Input artifact contract mismatch'
}

if (
    $SpeechRecord.kind -ne 'speech' -or
    $SpeechRecord.mime -ne 'audio/mpeg' -or
    [int64]$SpeechRecord.size_bytes -lt 1
) {
    throw 'Speech artifact contract mismatch'
}
```

用 Python stdlib 检查 PNG chunk：

```powershell
$PngReportPath =
    Join-Path $RunDir 'png-report.json'

@'
import hashlib
import json
import struct
import sys
from pathlib import Path

forbidden = {
    b"tEXt",
    b"zTXt",
    b"iTXt",
    b"eXIf",
}

report = []

for raw_path in sys.argv[2:]:
    path = Path(raw_path)
    data = path.read_bytes()

    if data[:8] != b"\x89PNG\r\n\x1a\n":
        raise SystemExit(
            f"not a PNG: {path.name}"
        )

    offset = 8
    forbidden_found = []

    while offset + 12 <= len(data):
        length = struct.unpack(
            ">I",
            data[offset:offset + 4],
        )[0]

        kind = data[offset + 4:offset + 8]
        end = offset + 12 + length

        if end > len(data):
            raise SystemExit(
                f"truncated PNG: {path.name}"
            )

        label = kind.decode("ascii", "replace")

        if kind in forbidden:
            forbidden_found.append(label)

        offset = end

        if kind == b"IEND":
            break

    if forbidden_found:
        raise SystemExit(
            "forbidden metadata chunks in "
            + path.name
            + ": "
            + ",".join(forbidden_found)
        )

    report.append({
        "filename": path.name,
        "sha256": hashlib.sha256(data).hexdigest(),
        "size_bytes": len(data),
        "forbidden_metadata_chunks": [],
    })

Path(sys.argv[1]).write_text(
    json.dumps(
        {
            "status": "pass",
            "screenshots": report,
        },
        separators=(",", ":"),
    ),
    encoding="utf-8",
)
'@ |
    python - `
        $PngReportPath `
        $ScreenshotPaths

if ($LASTEXITCODE -ne 0) {
    throw 'PNG metadata scan failed'
}

Copy-Item `
    $PngReportPath `
    "$EvidenceDir/screenshots.json" `
    -Force
```
