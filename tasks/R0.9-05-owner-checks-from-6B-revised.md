```powershell
function Set-ProcessSecret {
    param(
        [Parameter(Mandatory)]
        [string]$Name
    )

    $SecureValue =
        Read-Host "Enter $Name" -AsSecureString

    $Pointer =
        [Runtime.InteropServices.Marshal]::
            SecureStringToBSTR($SecureValue)

    try {
        $PlainValue =
            [Runtime.InteropServices.Marshal]::
                PtrToStringBSTR($Pointer)

        Set-Item `
            -Path "Env:$Name" `
            -Value $PlainValue
    } finally {
        [Runtime.InteropServices.Marshal]::
            ZeroFreeBSTR($Pointer)

        $PlainValue = $null
        $SecureValue = $null
    }
}
```

```powershell
$PngReportPath =
    Join-Path `
        $RunDir `
        'png-report.json'

@'
import hashlib
import json
import struct
import sys
from pathlib import Path

forbidden = {
    b"tEXt",
    b"zTXt",
    b"iTXt",
    b"eXIf",
}

report = []

for raw_path in sys.argv[2:]:
    path = Path(raw_path)
    data = path.read_bytes()

    if data[:8] != b"\x89PNG\r\n\x1a\n":
        raise SystemExit(f"not a PNG: {path.name}")

    offset = 8
    forbidden_found = []

    while offset + 12 <= len(data):
        length = struct.unpack(
            ">I",
            data[offset:offset + 4],
        )[0]

        kind = data[offset + 4:offset + 8]
        end = offset + 12 + length

        if end > len(data):
            raise SystemExit(
                f"truncated PNG: {path.name}"
            )

        label = kind.decode("ascii", "replace")

        if kind in forbidden:
            forbidden_found.append(label)

        offset = end

        if kind == b"IEND":
            break

    if forbidden_found:
        raise SystemExit(
            "forbidden metadata chunks in "
            + path.name
            + ": "
            + ",".join(forbidden_found)
        )

    report.append({
        "filename": path.name,
        "sha256": hashlib.sha256(data).hexdigest(),
        "size_bytes": len(data),
        "forbidden_metadata_chunks": [],
    })

Path(sys.argv[1]).write_text(
    json.dumps(
        {
            "status": "pass",
            "screenshots": report,
        },
        separators=(",", ":"),
    ),
    encoding="utf-8",
)
```
