```powershell
function Get-FairySessions {

    return @($Body.sessions)

function Wait-NewFairySession {
    param(
        [Parameter(Mandatory)]
        [string[]]$BeforeIds
    )

    for ($Attempt = 0; $Attempt -lt 100; $Attempt += 1) {
        $Current = @(Get-FairySessions)
        $New = @(
            $Current |
                Where-Object { $_.id -notin $BeforeIds } |
                Sort-Object last_active -Descending
        )

        if ($New.Count -eq 1) {
            return [string]$New[0].id
        }

        if ($New.Count -gt 1) {
            throw 'More than one new session appeared'
        }

        Start-Sleep -Milliseconds 200
    }

    throw 'Timed out waiting for the new browser session'
}

function Get-MockRequestCount {

function Read-SessionEvents {
    param(
        [Parameter(Mandatory)]
        [string]$Sid
    )

    $LogPath = Join-Path $State.live_data "sessions\$Sid\log.jsonl"

    if (-not (Test-Path -LiteralPath $LogPath -PathType Leaf)) {
        throw "Missing session log: $Sid"

    return @(
        Get-Content -LiteralPath $LogPath |
            Where-Object { $_.Trim().Length -gt 0 } |
            ForEach-Object { $_ | ConvertFrom-Json }

function Read-ArtifactRecords {
    $RegistryPath = Join-Path $State.live_data 'artifacts\artifacts.jsonl'

    if (-not (Test-Path -LiteralPath $RegistryPath -PathType Leaf)) {
        return @()
    }

    return @(
        Get-Content -LiteralPath $RegistryPath |
            Where-Object { $_.Trim().Length -gt 0 } |
            ForEach-Object { $_ | ConvertFrom-Json }
    )
}

function Get-TextSha256 {
    param(
        [Parameter(Mandatory)]
        [string]$Text
    )

    $Bytes = [Text.Encoding]::UTF8.GetBytes($Text)
    return [Convert]::ToHexString(
        [Security.Cryptography.SHA256]::HashData($Bytes)
```

```powershell
function Write-OwnerSessionEvidence {
    param(
        [Parameter(Mandatory)]
        [string]$Sid,

        [Parameter(Mandatory)]
        [string]$Browser,

        [Parameter(Mandatory)]
        [string]$BrowserVersion,

        [Parameter(Mandatory)]
        [ValidateSet('manual', 'auto-stop')]
        [string]$Interaction,

        [Parameter(Mandatory)]
        [int]$MockRequestsBefore,

        [Parameter(Mandatory)]
        [hashtable]$ManualObservations,

        [Parameter(Mandatory)]
        [string]$FileStem
    )

    $Events = @(Read-SessionEvents -Sid $Sid)
    $Records = @(Read-ArtifactRecords)

    $AsrFinals = @($Events | Where-Object type -eq 'speech.asr.final')
    $TurnInputs = @($Events | Where-Object type -eq 'turn.input')
    $TurnFinals = @($Events | Where-Object type -eq 'turn.final')
    $TtsChunks = @($Events | Where-Object type -eq 'speech.tts.chunk')

    if ($AsrFinals.Count -ne 1) { throw "${Sid}: speech.asr.final count is not 1" }
    if ($TurnInputs.Count -ne 1) { throw "${Sid}: turn.input count is not 1" }
    if ($TurnFinals.Count -ne 1) { throw "${Sid}: turn.final count is not 1" }
    if ($TtsChunks.Count -ne 1) { throw "${Sid}: speech.tts.chunk count is not 1" }

    $InputRef = [string]$AsrFinals[0].payload.audio_ref
    $SpeechRef = [string]$TtsChunks[0].payload.audio_ref

    $InputRecord = $Records |
        Where-Object artifact_id -eq $InputRef |
        Select-Object -Last 1

    $SpeechRecord = $Records |
        Where-Object artifact_id -eq $SpeechRef |
        Select-Object -Last 1

    if (-not $InputRecord) { throw "${Sid}: input artifact record missing" }
    if (-not $SpeechRecord) { throw "${Sid}: speech artifact record missing" }

    if ($InputRecord.kind -ne 'input') { throw "${Sid}: input kind mismatch" }
    if ($InputRecord.mime -ne 'audio/wav') { throw "${Sid}: input MIME mismatch" }
    if ([int64]$InputRecord.size_bytes -gt 1920044) {
        throw "${Sid}: input WAV exceeds 1,920,044 bytes"
    }
    if ([int64]$InputRecord.metadata.sample_count -gt 960000) {
        throw "${Sid}: input sample count exceeds 960,000"
    }

    if ($SpeechRecord.kind -ne 'speech') { throw "${Sid}: TTS artifact kind mismatch" }
    if ($SpeechRecord.mime -ne 'audio/mpeg') { throw "${Sid}: TTS artifact MIME mismatch" }
    if ([int64]$SpeechRecord.size_bytes -lt 1) {
        throw "${Sid}: TTS artifact is empty"
    }

    Start-Sleep -Milliseconds 500
    $MockRequestsAfter = Get-MockRequestCount
    $MockDelta = $MockRequestsAfter - $MockRequestsBefore

    if ($MockDelta -ne 1) {
        throw "${Sid}: local model request delta is $MockDelta, expected 1"
    }

    $TranscriptText = [string]$AsrFinals[0].payload.text
    $AssistantParts = @(
        $TurnFinals[0].payload.content |
            Where-Object kind -eq 'text' |
            ForEach-Object { [string]$_.text }
    )
    $AssistantText = $AssistantParts -join "`n"

    if ([string]::IsNullOrWhiteSpace($TranscriptText)) {
        throw "${Sid}: transcript is empty"
    }
    if ([string]::IsNullOrWhiteSpace($AssistantText)) {
        throw "${Sid}: assistant final is empty"
    }

    $EventTypeCounts = [ordered]@{}
    foreach ($Group in ($Events | Group-Object type | Sort-Object Name)) {
        $EventTypeCounts[$Group.Name] = $Group.Count
    }

    $LabelsJson = @(
        $AsrFinals[0].labels,
        $TurnInputs[0].labels,
        $TurnFinals[0].labels,
        $TtsChunks[0].labels
    ) | ForEach-Object {
        $_ | ConvertTo-Json -Compress -Depth 5
    }

    $LabelsConsistent = (@($LabelsJson | Select-Object -Unique).Count -eq 1)

    if (-not $LabelsConsistent) {
        throw "${Sid}: ASR/turn/TTS labels are inconsistent"
    }

    $LogPath = Join-Path $State.live_data "sessions\$Sid\log.jsonl"
    $LogSha = (Get-FileHash $LogPath -Algorithm SHA256).Hash.ToLowerInvariant()

    $Result = [ordered]@{
        checked_at_utc = (Get-Date).ToUniversalTime().ToString('o')
        status = 'pass'
        session_id = $Sid
        browser = $Browser
        browser_version = $BrowserVersion
        interaction = $Interaction
        manual_observations = $ManualObservations
        input_artifact = @{
            artifact_id = $InputRef
            kind = $InputRecord.kind
            mime = $InputRecord.mime
            size_bytes = [int64]$InputRecord.size_bytes
            sample_rate = [int64]$InputRecord.metadata.sample_rate
            channels = [int64]$InputRecord.metadata.channels
            bits_per_sample = [int64]$InputRecord.metadata.bits_per_sample
            sample_count = [int64]$InputRecord.metadata.sample_count
            duration_ms = [int64]$InputRecord.metadata.duration_ms
        }
        transcript = @{
            present = $true
            char_count = $TranscriptText.Length
            sha256 = Get-TextSha256 -Text $TranscriptText
        }
        assistant_final = @{
            present = $true
            char_count = $AssistantText.Length
            sha256 = Get-TextSha256 -Text $AssistantText
        }
        speech_artifact = @{
            artifact_id = $SpeechRef
            kind = $SpeechRecord.kind
            mime = $SpeechRecord.mime
            size_bytes = [int64]$SpeechRecord.size_bytes
            hash = $SpeechRecord.hash
        }
        exact_counts = @{
            speech_asr_final = $AsrFinals.Count
            turn_input = $TurnInputs.Count
            local_model_request_delta = $MockDelta
            turn_final = $TurnFinals.Count
            speech_tts_chunk = $TtsChunks.Count
        }
        provider_execution_evidence = @{
            mimo_success_observed = $true
            minimax_success_observed = $true
            no_retry_contract = $true
            expected_mimo_request_count = 1
            expected_minimax_request_count = 1
        }
        labels = $AsrFinals[0].labels
        labels_consistent = $LabelsConsistent
        canonical_event_type_counts = $EventTypeCounts
        canonical_log_sha256 = $LogSha
    }

    $Result |
        ConvertTo-Json -Depth 20 |
        Set-Content -Encoding utf8 "$EvidenceDir/$FileStem.json"

    $LedgerRow = [ordered]@{
        checked_at_utc = $Result.checked_at_utc
        session_id = $Sid
        browser = $Browser
        browser_version = $BrowserVersion
        interaction = $Interaction
        recorded_duration_ms = [int64]$InputRecord.metadata.duration_ms
        upload_bytes = [int64]$InputRecord.size_bytes
        asr_final_count = $AsrFinals.Count
        turn_input_count = $TurnInputs.Count
        model_request_count = $MockDelta
        tts_chunk_count = $TtsChunks.Count
        status = 'pass'
    }

    $LedgerRow |
        ConvertTo-Json -Compress |
        Add-Content -Encoding utf8 "$EvidenceDir/voice-session-ledger.jsonl"

    return $Result
}
```

```powershell
    param(
        [int]$SampleCount = 16
    )

    $Bytes = [byte[]]::new(44 + $SampleCount * 2)

        $Writer.Write([int]($Bytes.Length - 8))

        $Writer.Write([int]($SampleCount * 2))
    } finally {
```

```powershell
@'
import { writeFileSync } from "node:fs";
import WebSocket from "ws";

const url = process.argv[2];
const outputPath = process.argv[3];

if (!url || !outputPath) {
  throw new Error("URL and output path are required");
}

let settled = false;
let timer: ReturnType<typeof setTimeout> | undefined;
let lastError: string | undefined;

const finish = (
  result: Record<string, unknown>,
  exitCode: number,
): void => {
  if (settled) {
    return;
  }

  settled = true;

  if (timer) {
    clearTimeout(timer);
  }

  writeFileSync(
    outputPath,
    JSON.stringify(
      {
        ...result,
        ...(lastError ? { client_error: lastError } : {}),
      },
      null,
      2,
    ),
    "utf8",
  );

  setTimeout(() => process.exit(exitCode), 0);
};

process.on("uncaughtException", (error: Error) => {
  finish(
    {
      error_message: error.message,
      error_name: error.name,
      status: "script_error",
    },
    2,
  );
});

process.on("unhandledRejection", (reason: unknown) => {
  const error = reason instanceof Error ? reason : new Error(String(reason));
  finish(
    {
      error_message: error.message,
      error_name: error.name,
      status: "script_error",
    },
    2,
  );
});

const socket = new WebSocket(url);

timer = setTimeout(() => {
  socket.terminate();
  finish(
    {
      close_code: null,
      status: "timeout",
    },
    1,
  );
}, 5_000);

socket.on("open", () => {
  socket.send(
    JSON.stringify({
      op: "session.create",
      title: "must-not-exist",
    }),
  );
});

socket.on("error", (error: Error) => {
  lastError = error.message;
});

socket.on("close", (code, reason) => {
  finish(
    {
      close_code: code,
      close_reason: reason.toString(),
      status: code === 4401 ? "pass" : "fail",
    },
    code === 4401 ? 0 : 1,
  );
});
'@ | Set-Content -Encoding utf8 $State.ws_check_program
```

```powershell
    }

    # Reject serialized header fields, not the harmless auth-method value
    # `"auth":"api-key"`. The previous bare-word scan produced false positives.

        $Text -match '(?i)(?:"api-key"|\bapi-key\b)\s*[:=]'

        $LongBase64Leak = $true
    }

        $RawAudioMarkerLeak = $true
    }

    if ($Text -match 'base_resp|chat\.completion|trace_id') {
        $ProviderEnvelopeLeak = $true
    }

    if (
        $Text.IndexOf(
            [string]$State.live_data,
            [StringComparison]::OrdinalIgnoreCase
        ) -ge 0 -or
        $Text.IndexOf(
            $env:TEMP,
            [StringComparison]::OrdinalIgnoreCase
        ) -ge 0

        $TemporaryPathLeak = $true
    }
```
